```javascript
function fib(n) {
    if (n <= 2) {return 1;}
    return fib(n-1) + fib(n-2);
}
//or (my one liner)
const fib = (n, memo={2:1, 1:1}) => memo[n] || (memo[n] = fib(n-1, memo) + fib(n-2, memo))
// Try time complexity of fib(50)

console.assert(fib(6) == 8)
console.assert(fib(7) == 13)
console.assert(fib(8) == 21)
```
